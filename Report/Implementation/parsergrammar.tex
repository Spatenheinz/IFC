\begin{grammar}
<imp> ::= <quant> <imp'>

<imp'> ::= '$\Rightarrow$' <quant> <imp'> | $\epsilon$

<quant> ::= '$\forall$' <vname> '.' <quant>
\alt '$\exists$' <vname> '.' <quant>
\alt <or>

<or> ::= <and> '$\vee$' <or> | <and>

<and> ::= <neg> '$\wedge$' <and> | <neg>

<neg> ::= '$\sim$' <factor> | <factor>

<factor> ::=  <bexp> | '(' <imp> ')'
\end{grammar}
